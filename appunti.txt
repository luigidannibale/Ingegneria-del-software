-----------------------------------------------------Definizioni
Cadenza di gioco:
	Sia il tempo iniziale : x e il tempo incrementale : y,
	se fai una mossa entro y al tuo tempo viene aggiunto y.
	X decorre con il naturale passare del tempo.

Caratteristiche di una partita:
	- Cadenza di gioco
	- Squadra, la scelta delle pedine bianche e nere (casuale,bianco,nero)
	- Se partita classificata

Profilo di un giocatore:
	- Nome
	- Cognome
	- Username
	- Password
	- Punteggio elo

Impostazioni del sistema:
	- Colore scacchiera
	- Stile pedine

-------------------------------------------------------Funzioni
Matchmaking casuale:
	Siano giocatori g1,g2, che hanno entrambi cercato partita con cadenza di gioco c, (e |g1-g2| <= 100)
	crea partita tra g1 e g2 con cadenza di gioco c.

-------------------------------------------------------DB

CREATE TYPE Esito AS ENUM("B","N","P")
CREATE TYPE Motivo AS ENUM("")
CREATE TYPE Chessboard_style AS ENUM("blue","brown","black")
CREATE TYPE Pieces_style AS ENUM("neo","pixel")
CREATE TABLE User( 
	Username	PRIMARY KEY	Varchar(20) NOT NULL,
	Nome 				Varchar(20),
	Cognome 			Varchar(25),
	PuntiElo 			Integer NOT	NULL
	)


CREATE TABLE UserPreference( 
	User 	PRIMARY KEY Varchar(20) NOT NULL ,
	Tema 		Integer NOT NULL ,

	FOREIGN KEY Tema REFERENCES Temi(ID)
	FOREIGN KEY User REFERENCES User(Username) ON DELETE CASCADE
	)

CREATE TABLE Theme(
	ID					SERIAL PRIMARY KEY	NOT NULL,
	ColoreScacchiera	Integer NOT NULL,
	TemaPedine			Integer NOT NULL
)

CREATE TABLE Game (
	ID				SERIAL PRIMARY KEY	NOT NULL,
	White				Varchar(20) NOT NULL,
	Black				Varchar(20) NOT NULL,
	Duration			Int NOT NULL,
	Increment			Int NOT NULL,
	MosseGiocate		Varchar NOT NULL,
	Esito				Esito NOT NULL,
	Motivo				Motivo NOT NULL,

	FOREIGN KEY White REFERENCES User(Username),
	FOREIGN KEY Black REFERENCES User(Username),
	)

CREATE TABLE Logs(
	ID				SERIAL PRIMARY KEY	NOT NULL,
	Azione				Varchar NOT NULL,
	Timestamp			Timestamp NOT NULL 	
	)
